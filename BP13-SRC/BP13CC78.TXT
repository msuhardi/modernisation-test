      *                        IN RPT C; PRINT CASES W/ ARR-MTH = 0   *         
      *                        IN RPT C INSTEAD OF RPT A              *         
      * BP133091 RB12 20071017 EXPAND BP13FD00 FROM 1200 TO 1600 BYTES*

       PROCEDURE DIVISION.                                                      
      *===================*                                                     
       0000-MAIN-LOGIC.                                                         
      *===================*                                                     
                                                                                
           PERFORM 1000-INITIALIZATION   THRU 1000-EXIT.                        
           PERFORM 3000-READ-F125        THRU 3000-EXIT.                        
           PERFORM 2000-READ-FD00        THRU 2000-EXIT.                        
           PERFORM 4000-PROCESS          THRU 4000-EXIT                         
                   UNTIL EOF-OF-FD00.                                           
           PERFORM 9000-FINALIZATION     THRU 9000-EXIT.                        
                                                                                
       0000-EXIT.                                                               
           EXIT.                                                                
                                                                                
      *====================*                                                    
       1000-INITIALIZATION.                                                     
      *====================*                                                    
                                                                                
           OPEN INPUT  BP13F125                                                 
                       BP13FD00                                                 
                       BP13K800                                                 
                OUTPUT P13FD00A                                                 
                       P13FD00B                                                 
                       P13F125A                                                 
                       P13F125B                                                 
                       P13LC78A                                                 
                       P13LC78B                                                 
                       P13LC78C.                                                
                                                                                
           MOVE 'NO'     TO WS-EOF-F125                                         
                            WS-EOF-FD00.                                        
           MOVE ZEROES   TO WS-AMT-DIFF                                         
                            CNT-PRT-CASES-RPTA                                  
                            CNT-PRT-CASES-RPTB                                  
                            CNT-PRT-CASES-RPTC                                  
                            CNT-PAGE-RPTA                                       
                            CNT-PAGE-RPTB                                       
                            CNT-PAGE-RPTC                                       
                            WS-PRT-F125-FILEA                                   
                            WS-PRT-F125-FILEB                                   
                            WS-PRT-F125-FILEC                                   
                            WS-PRT-FD00-FILEA                                   
                            WS-PRT-FD00-FILEB                                   
                            WS-PRT-FD00-FILEC.                                  
                                                                                
           MOVE FUNCTION CURRENT-DATE TO WS-DATE.                               
           STRING WS-DATE(7:2) '/' WS-DATE(5:2) '/' WS-DATE(1:4)                
              DELIMITED BY SIZE INTO L125-DATE.                                 
                                                                                
       1000-EXIT.                                                               
           EXIT.                                                                
                                                                                
      *====================*                                                    
       2000-READ-FD00.                                                          
      *====================*                                                    
                                                                                
           READ BP13FD00 AT END MOVE 'YES' TO WS-EOF-FD00                       
                                MOVE HIGH-VALUE TO FD00-NUM-REGN                
               GO TO 2000-EXIT.                                                 
            ADD 1 TO WS-FD00-READ.                                              
                                                                                
       2000-EXIT.                                                               
           EXIT.                                                                
                                                                                
      *====================*                                                    
       3000-READ-F125.                                                          
      *====================*                                                    
                                                                                
           READ BP13F125 AT END MOVE 'YES' TO WS-EOF-F125                       
                                MOVE HIGH-VALUE TO F125-NUM-REGN                
               GO TO 3000-EXIT.                                                 
            ADD 1 TO WS-F125-READ.                                              
                                                                                
       3000-EXIT.                                                               
           EXIT.                                                                
                                                                                
      *====================*                                                    
       4000-PROCESS.                                                            
      *====================*                                                    
                                                                                
           MOVE ZEROES        TO WS-FD00-NRIC-CNT                               
                                 WS-F125-NRIC-CNT.                              
                                                                                
           IF FD00-NUM-MTH-INSTAL NOT NUMERIC                                   
              MOVE ZEROES      TO FD00-NUM-MTH-INSTAL                           
           END-IF.                                                              
                                                                                
           IF FD00-AMT-INSTAL NOT NUMERIC                                       
              MOVE ZEROES      TO FD00-AMT-INSTAL                               
           END-IF.                                                              
                                                                                
           IF FD00-AMT-INSTAL-CPF1 NOT NUMERIC                                  
              MOVE ZEROES      TO FD00-AMT-INSTAL-CPF1                          
           END-IF.                                                              
                                                                                
           IF FD00-AMT-INSTAL-CPF2 NOT NUMERIC                                  
              MOVE ZEROES      TO FD00-AMT-INSTAL-CPF2                          
           END-IF.                                                              
                                                                                
           IF FD00-AMT-INSTAL-CPF3 NOT NUMERIC                                  
              MOVE ZEROES      TO FD00-AMT-INSTAL-CPF3                          
           END-IF.                                                              
                                                                                
           IF FD00-AMT-INSTAL-CPF4 NOT NUMERIC                                  
              MOVE ZEROES      TO FD00-AMT-INSTAL-CPF4                          
           END-IF.                                                              
                                                                                
           IF FD00-AMT-INSTAL-CASH NOT NUMERIC                                  
              MOVE ZEROES      TO FD00-AMT-INSTAL-CASH                          
           END-IF.                                                              
           IF FD00-NUM-REGN = F125-NUM-REGN                                     
              PERFORM 4500-READ-K800      THRU 4500-EXIT                        
              IF K800-NUM-NRIC1 NOT = SPACES AND LOW-VALUES AND ZEROES          
                 ADD 1        TO WS-FD00-NRIC-CNT                               
              END-IF                                                            
              IF K800-NUM-NRIC2 NOT = SPACES AND LOW-VALUES AND ZEROES          
                 ADD 1        TO WS-FD00-NRIC-CNT                               
              END-IF                                                            
              IF K800-NUM-NRIC3 NOT = SPACES AND LOW-VALUES AND ZEROES          
                 ADD 1        TO WS-FD00-NRIC-CNT                               
              END-IF                                                            
              IF K800-NUM-NRIC4 NOT = SPACES AND LOW-VALUES AND ZEROES          
                 ADD 1        TO WS-FD00-NRIC-CNT                               
              END-IF                                                            
              PERFORM 5000-GET-F125-RECS  THRU 5000-EXIT                        
              PERFORM 2000-READ-FD00      THRU 2000-EXIT                        
           ELSE                                                                 
              IF FD00-NUM-REGN > F125-NUM-REGN                                  
                 PERFORM 6600-WRITE-NOMATCH-F125 THRU 6600-EXIT                 
                 PERFORM 3000-READ-F125          THRU 3000-EXIT                 
              ELSE                                                              
                 IF FD00-NUM-REGN NOT = SPACES AND LOW-VALUES AND ZEROES        
                    PERFORM 6500-WRITE-NOMATCH-FD00 THRU 6500-EXIT              
                    PERFORM 4100-MOVE-FD00-DATA     THRU 4100-EXIT              
                    IF FD00-AMT-INSTAL-CASH > 0                                 
                       PERFORM VARYING WS-CTR FROM 1 BY 1                       
                          UNTIL WS-CTR > 4                                      
                          MOVE ZEROES      TO WS-AMT-RCPT(WS-CTR)               
                          MOVE ZEROES      TO WS-NUM-DIFF(WS-CTR)               
                       END-PERFORM                                              
                       PERFORM 6510-PRINT-P13LC78A-GIRO THRU 6510-EXIT          
                    ELSE                                                        
                       IF FD00-NUM-MTH-INSTAL > ZEROES                          
                          ADD 1       TO WS-PRT-FD00-FILEB                      
                          MOVE BP13FD00-REC TO P13FD00A-REC                     
                          WRITE P13FD00A-REC                                    
                          MOVE SPACES TO WS-PRT-RPTA-SW                         
                          PERFORM 6100-PRINT-P13LC78A  THRU 6100-EXIT           
                       ELSE                                                     
                          MOVE SPACES TO WS-PRT-RPTC-SW                         
                          PERFORM 6300-PRINT-P13LC78C  THRU 6300-EXIT           
                       END-IF                                                   
                    END-IF                                                      
                 END-IF                                                         
                 PERFORM 2000-READ-FD00          THRU 2000-EXIT                 
              END-IF                                                            
           END-IF.                                                              
                                                                                
       4000-EXIT.                                                               
           EXIT.                                                                
                                                                                
      *====================*                                                    
       4100-MOVE-FD00-DATA.                                                     
      *====================*                                                    
                                                                                
           PERFORM VARYING WS-CTR FROM 1 BY 1 UNTIL WS-CTR > 4                  
              MOVE SPACES      TO WS-NUM-NRIC(WS-CTR)                           
              MOVE ZEROES      TO WS-AMT-INST(WS-CTR)                           
              MOVE ZEROES      TO WS-AMT-RCPT(WS-CTR)                           
              MOVE ZEROES      TO WS-NUM-DIFF(WS-CTR)                           
              MOVE SPACES      TO WS-SIGN(WS-CTR)                               
           END-PERFORM.                                                         
                                                                                
           PERFORM 4500-READ-K800      THRU 4500-EXIT.                          
           MOVE ZEROES         TO WS-CTR.                                       
           IF FD00-AMT-INSTAL-CPF1 > 0                                          
              ADD 1 TO WS-CTR                                                   
              MOVE K800-NUM-NRIC1   TO WS-NUM-NRIC(WS-CTR)                      
              MOVE FD00-AMT-INSTAL-CPF1                                         
                                    TO WS-AMT-INST(WS-CTR)                      
           END-IF.                                                              
                                                                                
           IF FD00-AMT-INSTAL-CPF2 > 0                                          
              ADD 1 TO WS-CTR                                                   
              MOVE K800-NUM-NRIC2   TO WS-NUM-NRIC(WS-CTR)                      
              MOVE FD00-AMT-INSTAL-CPF2                                         
                                    TO WS-AMT-INST(WS-CTR)                      
           END-IF.                                                              
                                                                                
           IF FD00-AMT-INSTAL-CPF3 > 0                                          
              ADD 1 TO WS-CTR                                                   
              MOVE K800-NUM-NRIC3   TO WS-NUM-NRIC(WS-CTR)                      
              MOVE FD00-AMT-INSTAL-CPF3                                         
                                    TO WS-AMT-INST(WS-CTR)                      
           END-IF.                                                              
                                                                                
           IF FD00-AMT-INSTAL-CPF4 > 0                                          
              ADD 1 TO WS-CTR                                                   
              MOVE K800-NUM-NRIC4   TO WS-NUM-NRIC(WS-CTR)                      
              MOVE FD00-AMT-INSTAL-CPF4                                         
                                    TO WS-AMT-INST(WS-CTR)                      
           END-IF.                                                              
                                                                                
       4100-EXIT.                                                               
           EXIT.                                                                
                                                                                
      *====================*                                                    
       4500-READ-K800.                                                          
      *====================*                                                    
                                                                                
           MOVE FD00-NUM-REGN      TO K800-NUM-REGN.                            
           READ BP13K800.                                                       
                                                                                
           IF K800-STATUS = ZEROS                                               
              ADD 1 TO WS-K800-READ                                             
           ELSE
