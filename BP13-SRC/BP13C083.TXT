       0000-EXIT.                                                               
           EXIT.

      *=============*                                                           
       3000-PROCESS.                                                            
      *=============*                                                           
                                                                                
           IF F730-CDE-HOUSEHOLD = SPACES OR LOW-VALUES                         
              PERFORM 4000-PRINT-P13L083A  THRU 4000-EXIT                       
           END-IF.                                                              
                                                                                
           IF F730-CDE-CATEGORY = SPACES OR LOW-VALUES                          
              PERFORM 4050-PRINT-P13L083B  THRU 4050-EXIT                       
           END-IF.                                                              
                                                                                
           IF F730-NUM-MTH-RESIDE > 60 AND                                      
              F730-NUM-5YR-OCCUPN = 'N'                                         
              PERFORM 4100-PRINT-P13L083C  THRU 4100-EXIT                       
           END-IF.                                                              
                                                                                
           IF F730-CDE-HOUSEHOLD = 'T'                                          
              PERFORM 4150-PRINT-P13L083D  THRU 4150-EXIT                       
           END-IF.                                                              
                                                                                
           IF  F730-NUM-5YR-OCCUPN = 'Y' AND                                    
              (F730-NUM-MTH-RESIDE NOT = 0 AND                                  
               F730-NUM-MTH-RESIDE <= 60)                                       
               PERFORM 4200-PRINT-P13L083E  THRU 4200-EXIT                      
           END-IF.                                                              
                                                                                
           IF F730-CDE-ELIG-SCH = 'PAS' OR 'PAA' OR 'NCS'                       
              PERFORM 4250-PRINT-P13L083F  THRU 4250-EXIT                       
           END-IF.                                                              
                                                                                
           IF (F730-CDE-RESIDENT-HA1 = '1' OR 'A' OR 'T') AND                   
              (F730-NUM-MCPS = 'C' OR 'P') AND                                  
              (F730-NUM-RES-NRIC1 = SPACES OR LOW-VALUES)                       
              PERFORM 4300-PRINT-P13L083G  THRU 4300-EXIT                       
           END-IF.                                                              
                                                                                
           IF (F730-CDE-RESIDENT-HA1 = SPACES OR LOW-VALUES) AND                
              (F730-NUM-MCPS = 'C' OR 'P')                                      
               IF (F730-NUM-RES-NRIC1 NOT = SPACES AND LOW-VALUES) OR           
                  (F730-NUM-RES-NRIC2 NOT = SPACES AND LOW-VALUES)              
                   PERFORM 4350-PRINT-P13L083H  THRU 4350-EXIT                  
               END-IF                                                           
           END-IF.                                                              
                                                                                
           IF (F730-CDE-NT1 = SPACES OR LOW-VALUES) OR                          
              (F730-CDE-FLAT-TYPE = SPACES OR LOW-VALUES)                       
               PERFORM 4400-PRINT-P13L083I  THRU 4400-EXIT                      
           END-IF.                                                              
                                                                                
           IF (F730-CDE-ALLOC-SCH = SPACES OR LOW-VALUES) AND                   
              (F730-CDE-ELIG-SCH = SPACES OR LOW-VALUES)                        
               PERFORM 4450-PRINT-P13L083J  THRU 4450-EXIT                      
           END-IF.                                                              
                                                                                
           IF F730-CDE-REQUEST-STATUS = 'C'                                     
              PERFORM 4500-PRINT-P13L083K  THRU 4500-EXIT                       
           END-IF.                                                              
                                                                                
           IF F730-NUM-PORT-ELIG-TAG = 'P'                                      
              PERFORM 4550-PRINT-P13L083L  THRU 4550-EXIT                       
           END-IF.                                                              
                                                                                
           IF (F730-NUM-MCPS = 'C' OR 'P') AND                                  
              (F730-CDE-RESIDENT-HA1 = 'P' OR 'H')                              
               PERFORM 4600-PRINT-P13L083M  THRU 4600-EXIT                      
           END-IF.                                                              
                                                                                
           IF F730-CDE-ALLOC-SCH = 'TCP'                                        
              PERFORM 4650-PRINT-P13L083N  THRU 4650-EXIT                       
           END-IF.                                                              
                                                                                
           PERFORM 2000-READ-F730    THRU 2000-EXIT.                            
                                                                                
       3000-EXIT.                                                               
           EXIT.
